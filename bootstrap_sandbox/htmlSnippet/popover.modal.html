Trigger Popover w/ Javascript

		<div class="container">   
			<button id="dismissible" class="btn btn-secondary" data-toggle="popover" data-placement="top" title="Popover Title"
			data-content="Lorem ipsum dolor sit amet, consectetur adipisicing elit." data-trigger="focus">
			  Toggle popover
			</button>		
			
			<button class="btn btn-dark" onClick="showPopover()">Show Popover</button>
			<button class="btn btn-light" onClick="hidePopover()">Hide Popover</button>
			<button class="btn grey" onClick="togglePopover()">Toggle Popover</button>
		</div>
			
			
			<script>
				// Init Popovers
				$('[data-toggle="popover"]').popover();

				function showPopover(){
				  $('#dismissible').popover('show');
				}

				function hidePopover(){
				  $('#dismissible').popover('hide');
				}

				function togglePopover(){
				  $('#dismissible').popover('toggle');
				}

				//Popover Events
				$('#dismissible').on('show.bs.popover', () => {
				  console.log("Popover Show")
				});

				$('#dismissible').on('shown.bs.popover', () => {
				  console.log("Popover Shown")
				});

				$('#dismissible').on('hide.bs.popover', () => {
				  console.log("Popover Hide")
				});

				$('#dismissible').on('hidden.bs.popover', () => {
				  console.log("Popover hidden")
				})
			</script>
