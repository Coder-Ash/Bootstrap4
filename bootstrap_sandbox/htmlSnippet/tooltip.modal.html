Trigger Tool Tip w/ Javascript

			<div class="container">  
				<button id="hello" class="btn btn-primary" data-toggle="tooltip" data-placement="top" data-html="true" 
					title="<h3>Hello World</h3><p>Lorem ipsum dolor sit amet.</p>">Tooltip With HTML</button>
					
				<button class="btn btn-secondary" onClick="showTooltip()">
				  Show Tooltip
				</button>

				<button class="btn btn-secondary" onClick="hideTooltip()">
				  Hide Tooltip
				</button>

				<button class="btn btn-secondary" onClick="toggleTooltip()">
				  Toggle Tooltip
				</button>
			</div>
			
			
			<script>
				// Init Tooltips
				$('[data-toggle="tooltip"]').tooltip();

				function showTooltip(){
				  $('#hello').tooltip('show');
				}

				function hideTooltip(){
				  $('#hello').tooltip('hide');
				}

				function toggleTooltip(){
				  $('#hello').tooltip('toggle');
				}

				//Tooltip Events
				$('#hello').on('show.bs.tooltip', () => {
				  console.log("Tooltip Show")
				});

				$('#hello').on('shown.bs.tooltip', () => {
				  console.log("Tooltip Shown")
				});

				$('#hello').on('hide.bs.tooltip', () => {
				  console.log("Tooltip Hide")
				});

				$('#hello').on('hidden.bs.tooltip', () => {
				  console.log("Tooltip hidden")
				});
			</script>
