			$("#crdTbLnk").click(()=> {
			  $("#crdTbActivate").removeClass("active");
			  $("#crdTbDisabled").removeClass("active");
			  $("#crdTbLnk").addClass("active");
			  $("#crdBdyLnk").removeClass("d-none");
			  $("#crdBdyActivate").addClass("d-none");
			  $("#crdBdyDisabled").addClass("d-none");
			});

			$("#crdTbActivate").click(()=> {
			  $("#crdTbLnk").removeClass("active");
			  $("#crdTbDisabled").removeClass("active");
			  $("#crdTbActivate").addClass("active");
			  $("#crdBdyActivate").removeClass("d-none");
			  $("#crdBdyLnk").addClass("d-none");
			  $("#crdBdyDisabled").addClass("d-none");
			});

			$("#crdTbDisabled").click(()=> {
			  $("#crdTbLnk").removeClass("active");
			  $("#crdTbActivate").removeClass("active");
			  $("#crdTbDisabled").addClass("active");
			  $("#crdBdyDisabled").removeClass("d-none");
			  $("#crdBdyLnk").addClass("d-none");
